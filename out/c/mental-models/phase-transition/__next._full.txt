<div class="concept" id="phase">
    <div class="concept-header">
      <div class="concept-num">03</div>
      <div class="concept-title-block">
        <h2>相变 · Phase Transition</h2>
        <div class="concept-subtitle">统计物理 · 用于发现系统临界点</div>
      </div>
      <span class="domain-badge tag-phys">物理</span>
    </div>

    <div class="block">
      <div class="block-label">核心直觉</div>
      <div class="formula">水在 99°C → 100°C 发生质变，不是量变
Agent 系统在某个临界点发生质变，不是线性退化

关键：找到你系统的相变点，在它之前设置硬限制
永远不要做线性外推（"再多加一点应该没问题"）</div>
    </div>

    <div class="block">
      <div class="block-label">Claude Code 中的三类相变点</div>
      <table class="decision-table">
        <tr><th>变量</th><th>相变表现</th><th>典型临界点</th><th>处理方式</th></tr>
        <tr>
          <td>Context 长度</td>
          <td>质量从 8/10 突然跌到 3/10</td>
          <td class="signal">~60-80% 窗口上限</td>
          <td class="action">硬截断 + 摘要压缩</td>
        </tr>
        <tr>
          <td>Prompt 复杂度</td>
          <td>模型开始幻觉、跳过指令</td>
          <td class="signal">约 15+ 条规则</td>
          <td class="action">分层 prompt + 子 agent</td>
        </tr>
        <tr>
          <td>并发 Agent 数量</td>
          <td>协调延迟 > 并行收益</td>
          <td class="signal">通信复杂度 O(n²) 拐点</td>
          <td class="action">分组 + 层级编排</td>
        </tr>
        <tr>
          <td>Retry 次数</td>
          <td>质量不再提升，开始循环</td>
          <td class="signal">3-5 次</td>
          <td class="action">切换策略，不增加次数</td>
        </tr>
      </table>
    </div>

      <div class="block-label">代码模式 — 主动探测相变点</div>

        <span>压力测试工具</span>
      </div>
<pre><span class="cm">// 主动探测 Context 相变点</span>
<span class="kw">async function</span> <span class="fn">findPhaseTransition</span>(
  basePrompt: <span class="kw">string</span>,
  targetTask: <span class="kw">string</span>,
) {
  <span class="kw">const</span> results: { tokens: <span class="kw">number</span>; quality: <span class="kw">number</span> }[] = [];
  <span class="kw">const</span> tokenLevels = [<span class="num">1000</span>, <span class="num">2000</span>, <span class="num">4000</span>, <span class="num">8000</span>, <span class="num">16000</span>, <span class="num">32000</span>];

  <span class="kw">for</span> (<span class="kw">const</span> tokens <span class="kw">of</span> tokenLevels) {
    <span class="kw">const</span> paddedPrompt = <span class="fn">padToTokens</span>(basePrompt, tokens);
    <span class="kw">const</span> quality = <span class="kw">await</span> <span class="fn">evaluateQuality</span>(paddedPrompt, targetTask);
    results.<span class="fn">push</span>({ tokens, quality });
    
    <span class="cm">// 检测相变：质量突然下降超过 30%</span>
    <span class="kw">if</span> (results.length > <span class="num">1</span>) {
      <span class="kw">const</span> prev = results[results.length - <span class="num">2</span>];
      <span class="kw">const</span> drop = (prev.quality - quality) / prev.quality;
      
      <span class="kw">if</span> (drop > <span class="num">0.3</span>) {
        console.<span class="fn">log</span>(<span class="str">`⚡ 相变点发现: ${tokens} tokens`</span>);
        console.<span class="fn">log</span>(<span class="str">`   质量从 ${prev.quality.toFixed(1)} 跌至 ${quality.toFixed(1)}`</span>);
        console.<span class="fn">log</span>(<span class="str">`   建议硬限制: ${prev.tokens} tokens`</span>);
        <span class="kw">break</span>;
      }
    }
  }

  <span class="kw">return</span> results;
}

<span class="cm">// 在系统中设置硬限制（不是软限制）</span>
<span class="kw">const</span> CONTEXT_HARD_LIMIT = <span class="num">6000</span>; <span class="cm">// 实测相变点 × 0.75</span>

<span class="kw">function</span> <span class="fn">safeContext</span>(messages: Message[]): Message[] {
  <span class="kw">const</span> totalTokens = <span class="fn">countTokens</span>(messages);
  <span class="kw">if</span> (totalTokens &lt;= CONTEXT_HARD_LIMIT) <span class="kw">return</span> messages;
  
  <span class="cm">// 不是线性截断，是结构化压缩</span>
  <span class="kw">return</span> [
    messages[<span class="num">0</span>],                          <span class="cm">// system prompt 保留</span>
    ...<span class="fn">summarizeMiddle</span>(messages.<span class="fn">slice</span>(<span class="num">1</span>, -<span class="num">4</span>)), <span class="cm">// 中间压缩</span>
    ...messages.<span class="fn">slice</span>(-<span class="num">4</span>),                  <span class="cm">// 最近 4 条保留</span>
  ];
}</pre>
